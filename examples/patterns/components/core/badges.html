Usage

<!-- Custom element -->
<status-badge data-type="success">Active</status-badge>

<!-- With dot indicator -->
<status-badge data-type="warning" data-dot>Pending</status-badge>

<!-- Small size -->
<status-badge data-type="error" data-size="small">Failed</status-badge>

<!-- Alternative: data-badge on any element -->
<span data-badge="info">New</span>
