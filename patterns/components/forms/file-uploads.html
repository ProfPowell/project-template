Usage

<!-- Basic file input -->
<form-field data-type="file">
  <label for="file">Upload file</label>
  <div data-file-input>
    <input type="file" id="file" />
    <button type="button" data-button="secondary">Choose File</button>
    <span data-file-name>No file chosen</span>
  </div>
</form-field>

<!-- Dropzone -->
<form-field data-type="file">
  <div data-dropzone>
    <input type="file" id="dropzone" multiple />
    <div data-dropzone-content>
      <p>Drop files here or click to upload</p>
    </div>
  </div>
</form-field>

<!-- With file restrictions -->
<input type="file" accept="image/*" multiple />
<input type="file" accept=".pdf,.doc,.docx" />
